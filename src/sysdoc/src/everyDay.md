 ####  二.数组中有一个数字出现的次数超过数组长度的一半，请找出这个数字。

 ```js

   if(item in pre){
    pre[item]++
  }else{
    pre[item] = 1 
  }
  return pre
},{})

function Result (index,numlength){

       }
    }
}
Result(index, numlength)
 ```
